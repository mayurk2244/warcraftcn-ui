## Installation

```bash
pnpm dlx shadcn@latest add https://warcraftcn.com/r/checkbox.json
```

## Usage

```tsx
import { Checkbox } from "@/components/ui/warcraftcn/checkbox";

export default function CheckboxExample () {
  return (
    <div className="flex gap-8">
      <Checkbox faction="default" label="Default" />
      <Checkbox faction="orc" label="Orc" />
      <Checkbox faction="elf" label="Elf" />
      <Checkbox faction="human" label="Human" />
      <Checkbox faction="undead" label="Undead" />
    </div>
  );
}
```
