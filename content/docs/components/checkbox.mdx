## Installation

```bash
pnpm dlx shadcn@latest add https://warcraftcn.com/r/checkbox.json
```

## Usage

```tsx
import { Checkbox } from "@/components/ui/warcraftcn/checkbox";
```

```tsx
<Checkbox faction="default" id="terms">
  Accept terms and conditions
</Checkbox>
```

### Faction Variants

```tsx
<Checkbox faction="human">Human theme</Checkbox>
<Checkbox faction="orc">Orc theme</Checkbox>
<Checkbox faction="elf">Elf theme</Checkbox>
<Checkbox faction="undead">Undead theme</Checkbox>
```

### Disabled State

```tsx
<Checkbox faction="default" disabled>
  Locked setting
</Checkbox>
```
